\section{ニューラルネットワーク}

\subsection{線形変換}

\begin{definition}{線形変換モデル}{linear}
$n,p\in\mathbb{N}$ とする．
重み行列 $W\in\mathbb{R}^{n\times p}$ および
入力 $x\in\mathbb{R}^{n}$ に対し，
写像 $f_W\colon\mathbb{R}^{n}\to\mathbb{R}^{p}$ を
\[
f_W(x)=W^\top x
\]
によって定める．
\end{definition}

\subsection{活性化関数}

線形変換の合成は再び線形変換となるため，
定義~\ref{def:linear}の線形変換を重ねてもモデルの表現力は向上しない．
非線形な活性化関数を導入することでこの制限を克服する．

\begin{definition}{活性化関数}{activation}
関数 $\sigma\colon\mathbb{R}\to\mathbb{R}$ を活性化関数と呼ぶ．
ベクトル $v=(v_1,\ldots,v_p)^\top\in\mathbb{R}^p$ に対し，
$\sigma$ の要素ごとの適用を
\[
\sigma(v) = \bigl(\sigma(v_1),\,\ldots,\,\sigma(v_p)\bigr)^\top
\]
と書く．
\end{definition}

\begin{example}{代表的な活性化関数}{activation_examples}
\leavevmode
\begin{itemize}
  \setlength{\itemsep}{2pt}
  \item ReLU：$\sigma(t) = \max(0,\, t)$
  \item シグモイド：$\sigma(t) = \dfrac{1}{1+e^{-t}}$
  \item $\tanh$：$\sigma(t) = \tanh(t)$
\end{itemize}
\end{example}

定義~\ref{def:linear}の線形変換モデルに活性化関数を適用すると，
\[
f_W(x) = \sigma(W^\top x)
\]
となる．$\sigma$ の非線形性により，
$f_W$ は線形変換では表現できない関数を近似できるようになる．

\subsection{多層ネットワーク}

定義~\ref{def:linear}の線形変換に活性化関数を合成したものを層と呼ぶ．

\begin{definition}{層}{layer}
重み行列 $W\in\mathbb{R}^{m\times n}$ および活性化関数 $\sigma$ に対し，
写像 $\ell_W\colon\mathbb{R}^{m}\to\mathbb{R}^{n}$ を
\[
\ell_W(x) = \sigma(W^\top x)
\]
によって定める．これを層と呼ぶ．
\end{definition}

この層を複数重ねることで，より複雑な関数を表現できる．

\begin{definition}{多層ネットワーク}{mlp}
隠れ層の数 $L\ge 1$ および各層の次元 $n_0,n_1,\ldots,n_{L+1}\in\mathbb{N}$ を与える．
$L$ 個の層 $\ell_{W_l}\colon\mathbb{R}^{n_{l-1}}\to\mathbb{R}^{n_l}$（$l=1,\ldots,L$）
および出力層の重み行列 $W_{L+1}\in\mathbb{R}^{n_L\times n_{L+1}}$ に対し，
入力 $x\in\mathbb{R}^{n_0}$ から出力への写像を
\[
f(x) = W_{L+1}^\top\,(\ell_{W_L}\circ\cdots\circ\ell_{W_1})(x)
\]
によって定める．すなわち，$h_0=x$ として
\[
h_l = \ell_{W_l}(h_{l-1}) = \sigma(W_l^\top h_{l-1})
\qquad (l=1,\ldots,L)
\]
と逐次的に計算し，$f(x)=W_{L+1}^\top h_L$ を出力とする．
出力層には活性化関数を適用しない．
\end{definition}

\begin{example}{$L=2$（ReLU + シグモイド）}{two_hidden}
$n_0=3,\, n_1=4,\, n_2=2,\, n_3=1$ とする．
第 1 層に ReLU，第 2 層にシグモイドを用いると，
入力 $x\in\mathbb{R}^3$ に対し
\[
\left\{
\begin{array}{c@{\;=\;}l}
h_1 & \ell_{W_1}(x) = \mathrm{ReLU}(W_1^\top x) \in \mathbb{R}^4, \\
h_2 & \ell_{W_2}(h_1) = \mathrm{sigmoid}(W_2^\top h_1) \in \mathbb{R}^2, \\
f(x) & W_3^\top h_2 \in \mathbb{R}
\end{array}
\right.
\]
となる．
ここで $W_1\in\mathbb{R}^{3\times 4}$，
$W_2\in\mathbb{R}^{4\times 2}$，
$W_3\in\mathbb{R}^{2\times 1}$ である．
\end{example}

各層の重み行列 $W_l$ に直交性制約 $W_l^\top W_l = I_{n_l}$ を課すことで，
各層での変換のスケールを一定に保つことができる．
本稿では以降，簡単のため単一の重み行列 $W\in\mathbb{R}^{n\times p}$
に対する最適化を扱う．

\subsection{損失関数}

滑らかな関数 $L\colon\mathbb{R}^{n\times p}\to\mathbb{R}$ を損失関数と呼ぶ．
これを最小化するような重み行列 $W$ を求めたい．
\[
\min_{W\in\mathbb{R}^{n\times p}}\, L(W)
\]

\begin{example}{平均二乗誤差}{mse}
訓練データ $\{(x_i,\, y_i)\}_{i=1}^{N}\subset\mathbb{R}^{n}\times\mathbb{R}^{p}$
に対し，平均二乗誤差（MSE）を
\[
L(W) = \frac{1}{N}\sum_{i=1}^{N}\bigl\|f_W(x_i)-y_i\bigr\|^2
\]
によって定める．これは回帰問題における代表的な損失関数である．
\end{example}

\begin{definition}{ユークリッド勾配}{egrad}
フロベニウス内積 $\langle A,B\rangle=\mathrm{tr}(A^\top B)$ に関して，
\[
dL(W)[Z] = \langle \nabla L(W),\, Z \rangle
\qquad (\forall\, Z\in\mathbb{R}^{n\times p})
\]
を満たす行列 $\nabla L(W)\in\mathbb{R}^{n\times p}$ を
$L$ の $W$ におけるユークリッド勾配と呼ぶ．
\end{definition}

\section{Stiefel 多様体}

重み行列 $W$ に直交性制約 $W^\top W=I_p$ を課すことを考える．
この制約は $W$ の特異値がすべて $1$ であることと同値であり，
$f_W$ による変換のスケールが一定に保たれる．

\begin{definition}{Stiefel 多様体}{stiefel}
$p\le n$ を仮定する．
Stiefel 多様体 $\mathrm{St}(p,n)$ を
\[
\mathrm{St}(p,n)
=
\{\, W\in\mathbb{R}^{n\times p}\mid W^\top W = I_p \,\}
\]
によって定義する．
\end{definition}

この制約のもとで，本稿では次の最適化問題を考える．
\[
\min_{W\in\mathrm{St}(p,n)}\, L(W).
\]

\subsection{接空間}

\begin{proposition}{}{tangent}
$W\in\mathrm{St}(p,n)$ における接空間は
\[
T_W\mathrm{St}(p,n)
=
\{\, Z\in\mathbb{R}^{n\times p}
\mid
W^\top Z + Z^\top W = 0
\,\}
\]
で与えられる．
\end{proposition}

\begin{proof}
$\mathcal{T}=\{\, Z\in\mathbb{R}^{n\times p}\mid W^\top Z + Z^\top W = 0\,\}$ とおく．

($T_W\mathrm{St}(p,n)\subset\mathcal{T}$)\\
$\mathrm{St}(p,n)$ 上の滑らかな曲線 $W(t)$ が $W(0)=W$ を満たすとする．
制約 $W(t)^\top W(t)=I_p$ の両辺を $t=0$ で微分すると
$\dot W(0)^\top W + W^\top\dot W(0)=0$
を得る．すなわち $\dot W(0)\in\mathcal{T}$ である．

($\mathcal{T}\subset T_W\mathrm{St}(p,n)$)\\
制約写像 $F\colon\mathbb{R}^{n\times p}\to\mathrm{Sym}(p)=\{\, S\in\mathbb{R}^{p\times p}\mid S=S^\top \,\}$ を
$F(W)=W^\top W - I_p$ で定める．
$F$ の微分は
\[
dF(W)[Z]=W^\top Z + Z^\top W
\]
で与えられる．任意の $S\in\mathrm{Sym}(p)$ に対し $Z=\tfrac{1}{2}WS$ とおくと
\[
dF(W)\bigl[\tfrac{1}{2}WS\bigr]
= \tfrac{1}{2}(W^\top W S + S W^\top W)
= S
\]
となるから，$dF(W)$ は全射である．
したがって $\mathrm{St}(p,n)=F^{-1}(0)$ は正則値の逆像であり，
$T_W\mathrm{St}(p,n)=\ker dF(W)=\mathcal{T}$ が成り立つ．
\end{proof}

\subsection{リーマン勾配}

\begin{definition}{リーマン勾配}{rgrad}
$W\in\mathrm{St}(p,n)$ において，
\[
dL(W)[Z] = \langle \mathrm{grad}\,L(W),\, Z \rangle
\qquad (\forall\, Z\in T_W\mathrm{St}(p,n))
\]
を満たす $\mathrm{grad}\,L(W)\in T_W\mathrm{St}(p,n)$ を
$L$ の $W$ におけるリーマン勾配と呼ぶ．
\end{definition}

ユークリッド勾配 $\nabla L(W)$ は $\mathbb{R}^{n\times p}$ 全体における最急降下方向であるが，
多様体上で動ける方向は接空間 $T_W\mathrm{St}(p,n)$ に限られる．
ユークリッド勾配には接空間の法線方向の成分が含まれており，
この成分は多様体上での $L$ の変化に寄与しない．
リーマン勾配はユークリッド勾配から法線成分を除いたものであり，
多様体上での正しい最急降下方向を与える．

\begin{remark}{直交分解}{orthogonal_decomp}
内積空間において，ノルム $1$ の元 $u$ が与えられたとき，
任意の元 $v$ は法線成分 $\langle u,v\rangle\, u$ と
接線成分 $v - \langle u,v\rangle\, u$ に直交分解できる．
この分解はベクトルに限らず，
フロベニウス内積を備えた行列空間など一般の内積空間で成り立つ．
\end{remark}

\begin{proposition}{リーマン勾配の計算}{rgrad_formula}
リーマン勾配は，ユークリッド勾配 $G=\nabla L(W)$ を用いて
\[
\mathrm{grad}\,L(W)
=
G - W\,\mathrm{sym}(W^\top G)
\]
で与えられる．ここで $\mathrm{sym}(A)=\tfrac{1}{2}(A+A^\top)$ である．
\end{proposition}

\begin{proof}
$G$ を接空間成分 $Z$ と法線成分 $WS$（$S$ は対称行列）に分解する：
\[
G=Z+WS.
\]
$W^\top W=I_p$ より $W^\top G = W^\top Z + S$ である．
接空間条件 $W^\top Z + Z^\top W=0$ から
$W^\top Z$ は歪対称であり，
\[
S
= \mathrm{sym}(W^\top G)
\]
と定まる．
したがって $Z=G-W\,\mathrm{sym}(W^\top G)$ を得る．
\end{proof}

\begin{example}{単位円上のリーマン勾配}{rgrad_circle}
$n=2,\, p=1$ の場合，$\mathrm{St}(1,2)$ は単位円 $S^1$ である．
$L(w)=(w_1)^2$ とし，
$a=\begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}\in S^1$
における勾配を考える．各成分で偏微分すると，ユークリッド勾配は
\[
\nabla L(w)
=\begin{pmatrix}\partial L/\partial w_1\\\partial L/\partial w_2\end{pmatrix}
=\begin{pmatrix}2w_1\\0\end{pmatrix}
\]
であるから，$w=a$ では
$\nabla L(a)=(\sqrt{2},\,0)^\top$ となる．
$a$ における接空間は
$T_{a} S^1 = \{z\in\mathbb{R}^2 \mid a^\top z = 0\}$
であり，ユークリッド勾配 $\nabla L(a)$ は接空間に属さない．
実際，$a^\top \nabla L(a) = \frac{1}{\sqrt{2}}\cdot\sqrt{2} = 1 \neq 0$ である．

注意~\ref{rem:orthogonal_decomp}の直交分解を
$\nabla L(a)$ に適用すると，
\[
\underbrace{\begin{pmatrix}\sqrt{2}\\0\end{pmatrix}}_{\nabla L(a)}
=
\underbrace{\begin{pmatrix}1/\sqrt{2}\\-1/\sqrt{2}\end{pmatrix}}_{\text{接線成分}}
+
\underbrace{\begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}}_{\text{法線成分}}
\]
となる．法線成分は円から離れる方向であり，
円上での $L$ の変化に寄与しない．
したがって，$-\nabla L(a)$ の方向に動くと円の外に出てしまい，
円上における最急降下方向とはならない．

一方，リーマン勾配は接線成分のみを取り出したものであり，
\[
\mathrm{grad}\,L(a)
= \nabla L(a) - a\,(a^\top \nabla L(a))
= \begin{pmatrix}\sqrt{2}\\0\end{pmatrix}
  - \begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}
= \begin{pmatrix}1/\sqrt{2}\\-1/\sqrt{2}\end{pmatrix}
\]
となる．これは命題~\ref{prop:rgrad_formula}の公式
$G - W\,\mathrm{sym}(W^\top G)$ において $p=1$ とした場合に一致する．
$-\mathrm{grad}\,L(a)$ の方向に進めば
円上に沿って $L$ が減少する．
\end{example}

\subsection{レトラクション}

リーマン勾配の方向に $W$ を更新すると，
$W - \eta\,\mathrm{grad}\,L(W)$ は一般に $\mathrm{St}(p,n)$ 上に留まらない．
そこで，接空間上の更新を多様体上に引き戻す写像が必要となる．

\begin{definition}{レトラクション}{retraction}
多様体 $\mathcal{M}$ 上の点 $x$ において，
滑らかな写像 $R_x\colon T_x\mathcal{M}\to\mathcal{M}$ が
以下の 2 条件を満たすとき，$R$ をレトラクションと呼ぶ：
\begin{enumerate}
\item $R_x(0_x) = x$，
\item $\displaystyle\frac{d}{dt}R_x(tZ)\bigg|_{t=0} = Z
  \qquad (\forall\, Z\in T_x\mathcal{M})$．
\end{enumerate}
\end{definition}

条件 1 はゼロ方向への更新が現在の点を保つことを，
条件 2 は微小な更新が接ベクトルの方向に沿うことを要請する．

\begin{definition}{QR 分解}{qr}
$\mathrm{rank}(A)=p$ を満たす行列 $A\in\mathbb{R}^{n\times p}$ に対し，
\[
A = QR,
\qquad Q\in\mathrm{St}(p,n),\quad R\in\mathbb{R}^{p\times p}
\]
を満たす直交行列 $Q$ と上三角行列 $R$（対角成分が正）の組が一意に存在する．
これを $A$ の QR 分解と呼び，$Q=\mathrm{qf}(A)$ と書く．
\end{definition}

\begin{example}{}{qr_example}
$n=3,\, p=2$ の場合：
\[
\underbrace{
\begin{pmatrix}
1 & 1 \\
1 & 0 \\
0 & 1
\end{pmatrix}
}_{A}
=
\underbrace{
\begin{pmatrix}
\frac{1}{\sqrt{2}} & \frac{1}{\sqrt{6}} \\[4pt]
\frac{1}{\sqrt{2}} & -\frac{1}{\sqrt{6}} \\[4pt]
0 & \frac{2}{\sqrt{6}}
\end{pmatrix}
}_{Q=\mathrm{qf}(A)}
\underbrace{
\begin{pmatrix}
\sqrt{2} & \frac{1}{\sqrt{2}} \\[4pt]
0 & \frac{3}{\sqrt{6}}
\end{pmatrix}
}_{R}.
\]
\end{example}

$\mathrm{qf}(A)\in\mathrm{St}(p,n)$ であるから，
更新後の行列に QR 分解を適用することで多様体上に戻すことができる．
すなわち，$W\in\mathrm{St}(p,n)$ および $Z\in T_W\mathrm{St}(p,n)$ に対し，
\[
R_W(Z) = \mathrm{qf}(W+Z)
\]
は定義~\ref{def:retraction}の意味でレトラクションとなる．
これを QR レトラクションと呼ぶ．

\section{アルゴリズム}

\subsection{勾配降下法}

制約のない場合，損失関数 $L$ の最小化にはユークリッド勾配を用いた
勾配降下法が基本的な手法となる．

\begin{definition}{勾配降下法}{gd}
初期値 $W_0\in\mathbb{R}^{n\times p}$ および学習率 $\eta>0$ を与える．
反復
\[
W_{k+1} = W_k - \eta\,\nabla L(W_k)
\qquad (k=0,1,2,\ldots)
\]
を勾配降下法と呼ぶ．
\end{definition}

しかし，勾配降下法では更新後の $W$ が $\mathrm{St}(p,n)$ 上に留まる保証がない．

\subsection{リーマン勾配降下法}

第2節で導入したリーマン勾配とレトラクションを組み合わせることで，
Stiefel 多様体上の最適化問題
\[
\min_{W\in\mathrm{St}(p,n)}\, L(W)
\]
を解くことができる．
アルゴリズムを Algorithm~\ref{alg:rsgd} にまとめる．

\begin{algorithm}{alg:rsgd}
\textbf{Stiefel 多様体上のリーマン勾配降下法} \\[4pt]
\textbf{Input:} 損失関数 $L$，学習率 $\eta > 0$，初期値 $W_0 \in \mathrm{St}(p,n)$ \\[4pt]
\textbf{for} $k = 0, 1, 2, \ldots$ \textbf{do}
\begin{enumerate}
  \setlength{\itemsep}{2pt}
  \item $G_k \gets \nabla L(W_k)$
    \hfill $\triangleright$ ユークリッド勾配（定義~\ref{def:egrad}）
  \item $\xi_k \gets G_k - W_k\,\mathrm{sym}(W_k^\top G_k)$
    \hfill $\triangleright$ リーマン勾配（命題~\ref{prop:rgrad_formula}）
  \item $W_{k+1} \gets \mathrm{qf}(W_k - \eta\,\xi_k)$
    \hfill $\triangleright$ QR レトラクション（定義~\ref{def:qr}）
\end{enumerate}
\textbf{end for}
\end{algorithm}

各ステップで QR 分解により $W_{k+1}\in\mathrm{St}(p,n)$ が保証されるため，
すべての反復を通じて直交性制約 $W_k^\top W_k = I_p$ が保たれる．
