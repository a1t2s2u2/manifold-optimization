\section{深層学習における最適化問題}

本節では，以降の議論に必要な基本的枠組みを定める．

\begin{definition}{線形変換モデル}{linear}
$n,p\in\mathbb{N}$ とする．
重み行列 $W\in\mathbb{R}^{n\times p}$ および
入力 $x\in\mathbb{R}^{n}$ に対し，
写像 $f_W\colon\mathbb{R}^{n}\to\mathbb{R}^{p}$ を
\[
f_W(x)=W^\top x
\]
によって定める．
\end{definition}

滑らかな関数 $L\colon\mathbb{R}^{n\times p}\to\mathbb{R}$ を損失関数と呼ぶ．これを最小化するような重み行列 $W$ を求めたい．
\[
\min_{W\in\mathbb{R}^{n\times p}}\, L(W)
\]

深層学習では，一般的に勾配降下法を用いてこの最適化問題を解く．

\begin{definition}{ユークリッド勾配}{egrad}
フロベニウス内積 $\langle A,B\rangle=\mathrm{tr}(A^\top B)$ に関して，
\[
dL(W)[Z] = \langle \nabla L(W),\, Z \rangle
\qquad (\forall\, Z\in\mathbb{R}^{n\times p})
\]
を満たす行列 $\nabla L(W)\in\mathbb{R}^{n\times p}$ を
$L$ の $W$ におけるユークリッド勾配と呼ぶ．
\end{definition}

\begin{definition}{勾配降下法}{gd}
初期値 $W_0\in\mathbb{R}^{n\times p}$ および学習率 $\eta>0$ を与える．
反復
\[
W_{k+1} = W_k - \eta\,\nabla L(W_k)
\qquad (k=0,1,2,\ldots)
\]
を勾配降下法と呼ぶ．
\end{definition}

\section{Stiefel 多様体上の最適化}

本節では，重み行列 $W$ に直交性制約 $W^\top W=I_p$ を課した最適化問題を考える．
この制約は $W$ の特異値がすべて $1$ であることと同値であり，
$f_W$ による変換のスケールが一定に保たれる．

\begin{definition}{Stiefel 多様体}{stiefel}
$p\le n$ を仮定する．
Stiefel 多様体 $\mathrm{St}(p,n)$ を
\[
\mathrm{St}(p,n)
=
\{\, W\in\mathbb{R}^{n\times p}\mid W^\top W = I_p \,\}
\]
によって定義する．
\end{definition}

この制約のもとで損失関数 $L$ に関する次の最適化問題を考える．
\[
\min_{W\in\mathrm{St}(p,n)}\, L(W).
\]

\begin{remark}{}{gremark}
$\mathrm{St}(p,n)$ は $\mathbb{R}^{n\times p}$ の閉部分多様体であり，
通常の勾配降下法による更新 $W-\eta\,\nabla L(W)$ は
一般に $\mathrm{St}(p,n)$ 上に留まらない．
したがって，多様体の幾何構造を考慮した最適化手法が必要となる．
\end{remark}

\subsection{接空間}

\begin{proposition}{}{tangent}
$W\in\mathrm{St}(p,n)$ における接空間は
\[
T_W\mathrm{St}(p,n)
=
\{\, Z\in\mathbb{R}^{n\times p}
\mid
W^\top Z + Z^\top W = 0
\,\}
\]
で与えられる．
\end{proposition}

\begin{proof}
$\mathcal{T}=\{\, Z\in\mathbb{R}^{n\times p}\mid W^\top Z + Z^\top W = 0\,\}$ とおく．

($T_W\mathrm{St}(p,n)\subset\mathcal{T}$)\\
$\mathrm{St}(p,n)$ 上の滑らかな曲線 $W(t)$ が $W(0)=W$ を満たすとする．
制約 $W(t)^\top W(t)=I_p$ の両辺を $t=0$ で微分すると
$\dot W(0)^\top W + W^\top\dot W(0)=0$
を得る．すなわち $\dot W(0)\in\mathcal{T}$ である．

($\mathcal{T}\subset T_W\mathrm{St}(p,n)$)\\
制約写像 $F\colon\mathbb{R}^{n\times p}\to\mathrm{Sym}(p)=\{\, S\in\mathbb{R}^{p\times p}\mid S=S^\top \,\}$ を
$F(W)=W^\top W - I_p$ で定める．
$F$ の微分は
\[
dF(W)[Z]=W^\top Z + Z^\top W
\]
で与えられる．任意の $S\in\mathrm{Sym}(p)$ に対し $Z=\tfrac{1}{2}WS$ とおくと
\[
dF(W)\bigl[\tfrac{1}{2}WS\bigr]
= \tfrac{1}{2}(W^\top W S + S W^\top W)
= S
\]
となるから，$dF(W)$ は全射である．
したがって $\mathrm{St}(p,n)=F^{-1}(0)$ は正則値の逆像であり，
$T_W\mathrm{St}(p,n)=\ker dF(W)=\mathcal{T}$ が成り立つ．
\end{proof}

\subsection{リーマン勾配}

\begin{definition}{リーマン勾配}{rgrad}
$W\in\mathrm{St}(p,n)$ において，
\[
dL(W)[Z] = \langle \mathrm{grad}\,L(W),\, Z \rangle
\qquad (\forall\, Z\in T_W\mathrm{St}(p,n))
\]
を満たす $\mathrm{grad}\,L(W)\in T_W\mathrm{St}(p,n)$ を
$L$ の $W$ におけるリーマン勾配と呼ぶ．
\end{definition}

\begin{proposition}{}{rgrad_formula}
リーマン勾配は，ユークリッド勾配 $G=\nabla L(W)$ を用いて
\[
\mathrm{grad}\,L(W)
=
G - W\,\mathrm{sym}(W^\top G)
\]
で与えられる．ここで $\mathrm{sym}(A)=\tfrac{1}{2}(A+A^\top)$ である．
\end{proposition}

\begin{proof}
$G$ を接空間成分 $Z$ と法線成分 $WS$（$S$ は対称行列）に分解する：
\[
G=Z+WS.
\]
$W^\top W=I_p$ より $W^\top G = W^\top Z + S$ である．
接空間条件 $W^\top Z + Z^\top W=0$ から
$W^\top Z$ は歪対称であり，
\[
S
= \mathrm{sym}(W^\top G)
\]
と定まる．
したがって $Z=G-W\,\mathrm{sym}(W^\top G)$ を得る．
\end{proof}

\subsection{レトラクション}

\begin{definition}{QR レトラクション}{retraction}
$W\in\mathrm{St}(p,n)$ および $Z\in T_W\mathrm{St}(p,n)$ に対し，
レトラクション $R_W\colon T_W\mathrm{St}(p,n)\to\mathrm{St}(p,n)$ を
\[
R_W(Z) = \mathrm{qf}(W+Z)
\]
によって定める．
ここで $\mathrm{qf}$ は QR 分解における直交因子を表す．
\end{definition}

\subsection{アルゴリズム}

以上を組み合わせることで，
Stiefel 多様体上の勾配降下法が次のように得られる．

学習率 $\eta>0$ とし，初期値 $W_0\in\mathrm{St}(p,n)$ を与える．
各反復 $k=0,1,2,\ldots$ において：
\[
\begin{aligned}
G_k &= \nabla L(W_k),\\
\xi_k &= G_k - W_k\,\mathrm{sym}(W_k^\top G_k),\\
W_{k+1} &= \mathrm{qf}(W_k - \eta\,\xi_k).
\end{aligned}
\]
