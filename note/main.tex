\begin{definition}[問題設定]
$n,p\in\mathbb{N}$ とする．$W\in\mathbb{R}^{n\times p}, x\in\mathbb{R}^{n}$ に対して
\[
f_W(x)=W^\top x
\]
を定める．
\end{definition}

\subsection{Stiefel 多様体上の最適化}

\begin{definition}[Stiefel 多様体]
$n,p\in\mathbb{N}$ を $p\le n$ を満たす整数とする．
実数体上の Stiefel 多様体 $\mathrm{St}(p,n)$ を
\[
\mathrm{St}(p,n)
=
\{\, W\in\mathbb{R}^{n\times p}\mid W^\top W = I_p \,\}
\]
によって定義する．
\end{definition}

$n,p\in\mathbb{N}$ とし，滑らかな関数 $L:\mathbb{R}^{n\times p}\to\mathbb{R}$ を与える．
Stiefel 多様体上の最適化問題を
\[
\min_{W\in\mathrm{St}(p,n)} L(W)
\]
と定める．

\paragraph{接空間}

\begin{proposition}[接空間の導出]
$W\in\mathrm{St}(p,n)$ における接空間は
\[
T_W\mathrm{St}(p,n)
=
\{\, Z\in\mathbb{R}^{n\times p}
\mid
W^\top Z + Z^\top W = 0
\,\}
\]
で与えられる．
\end{proposition}

\begin{proof}
制約関数 $F$ を
\[
F(W)=W^\top W-I_p
\]
とおく．
曲線 $W(t)$ が $\mathrm{St}(p,n)$ 上にあるならば、$F(W(t))=0$ が成り立つ．
これを $t$ で微分すると
\[
\frac{d}{dt}(W^\top W)
=
\dot W^\top W + W^\top \dot W = 0
\]
となる．
$Z=\dot W$ とおけば，
\[
W^\top Z + Z^\top W =0
\]
を得る．
\end{proof}

\paragraph{リーマン勾配}

\begin{definition}[ユークリッド勾配]
$L:\mathbb{R}^{n\times p}\to\mathbb{R}$ を滑らかな関数とする．
フロベニウス内積
\[
\langle A,B\rangle := \mathrm{tr}(A^\top B)
\]
に関して，
\[
dL(W)[Z]
=
\langle \nabla L(W), Z\rangle
\qquad
(Z\in\mathbb{R}^{n\times p})
\]
を満たす行列 $\nabla L(W)$ を
ユークリッド勾配という．
\end{definition}

\begin{remark}
フロベニウス内積は
\[
\langle A,B\rangle
=
\sum_{i=1}^{n}\sum_{j=1}^{p} A_{ij}B_{ij}
\]
で与えられ，
行列をベクトルとみなしたときの通常の内積に対応する．
\end{remark}

\begin{remark}
ユークリッド勾配 $\nabla L(W)$ は，
$W$ をどの方向に動かすと
$L$ が最も増加するかを表す行列である．
\end{remark}

\textbf{補足．}
フロベニウス内積は，行列を成分ごとに掛けて足し合わせたものであり，
$\mathbb{R}^{n\times p}$ を通常のユークリッド空間と同様に扱うための内積である．
すなわち，ベクトルの場合の内積
$\langle x,y\rangle=\sum_i x_i y_i$
を行列に拡張したものとみなせる．
また，$\mathrm{tr}(A^\top B)$ という形に書くことで，
成分表示を用いずに簡潔に記述できる．

\noindent
ユークリッド勾配 $\nabla L(W)$ は，
「$W$ をどの方向に動かすと $L$ が最も増加するか」
を表す行列であり，
通常の最急降下法では
\[
W \leftarrow W - \eta\,\nabla L(W)
\]
という更新に対応する．
ただし Stiefel 多様体上では，
この勾配をそのまま用いるのではなく，
接空間への射影を通してリーマン勾配が定まる．


$G$ を接空間へ射影したものを
Stiefel 多様体上の勾配（リーマン勾配）と呼び，
\[
\mathrm{grad}\,L(W)
=
G - W\,\mathrm{sym}(W^\top G)
\]
で与えられる．
ここで
\[
\mathrm{sym}(A)=\frac12(A+A^\top)
\]
である．

\begin{proof}[射影公式]
$G$ を接空間成分 $Z$ と法線成分 $WS$ に分解する：
\[
G=Z+WS.
\]
接空間条件 $W^\top Z+Z^\top W=0$ を満たすように
対称行列 $S$ を求めると，
\[
S=\mathrm{sym}(W^\top G)
\]
となる．
したがって
\[
Z=G-W\,\mathrm{sym}(W^\top G)
\]
を得る．
\end{proof}

\paragraph{更新則（Retract）}
学習率 $\eta>0$ に対して，
ユークリッド空間での更新
\[
\widetilde W = W - \eta\,\mathrm{grad}\,L(W)
\]
は一般に Stiefel 多様体上に留まらない．
そこで retraction を用いて
\[
W_{\text{new}}
=
\mathrm{qf}(\widetilde W)
\]
と定める．
ここで $\mathrm{qf}$ は QR 分解の直交因子を表す．

\paragraph{アルゴリズム}
以上より，Stiefel 多様体上の勾配降下法は
次の更新で与えられる：
\[
\begin{aligned}
G &= \nabla L(W),\\
G_R &= G - W\,\mathrm{sym}(W^\top G),\\
W &\leftarrow \mathrm{qf}(W - \eta G_R).
\end{aligned}
\]
