\section{ニューラルネットワーク}

ニューラルネットワークは，生物の脳における神経細胞（ニューロン）の仕組みを模した数理モデルであり，入力データから目標値への写像を
パラメータ付きの関数として表現する．
重み行列と呼ばれるパラメータを調整することで，入力に対して望ましい出力を返すように学習させることができる．
本節では，活性化関数を導入した上でモデルの基本単位である層を定義し，
それを複数重ねた多層ネットワークを構成する．

\subsection{活性化関数}

\begin{definition}{活性化関数}{activation}
関数 $\sigma\colon\mathbb{R}\to\mathbb{R}$ を活性化関数と呼ぶ．
ベクトル $v=(v_1,\ldots,v_n)^\top\in\mathbb{R}^n$ に対し，
$\sigma$ の要素ごとの適用を
\[
\sigma(v) = \bigl(\sigma(v_1),\,\ldots,\,\sigma(v_n)\bigr)^\top
\]
と書く．
\end{definition}

線形変換の合成は再び線形変換となるため，
層を複数重ねてより複雑な関数を表現できるよう，
活性化関数には通常，非線形な関数が用いられる．

\begin{example}{代表的な活性化関数}{activation_examples}
\leavevmode
\begin{itemize}
  \setlength{\itemsep}{6pt}
  \item ReLU：$\sigma(t) = \max(0,\, t)$，\quad
    $\sigma'(t) = \begin{cases}1 & (t>0)\\0 & (t\le 0)\end{cases}$
  \item シグモイド：$\sigma(t) = \dfrac{1}{1+e^{-t}}$，\quad
    $\sigma'(t) = \sigma(t)\bigl(1-\sigma(t)\bigr)$
  \item $\tanh$：$\sigma(t) = \tanh(t)$，\quad
    $\sigma'(t) = 1-\tanh^2(t)$
\end{itemize}
\end{example}

\subsection{層}

\begin{definition}{層}{layer}
重み行列 $W\in\mathbb{R}^{m\times n}$ および活性化関数 $\sigma$ に対し，
写像 $\ell_W\colon\mathbb{R}^{m}\to\mathbb{R}^{n}$ を
\[
\ell_W(x) = \sigma(W^\top x)
\]
によって定める．これを層と呼ぶ．
\end{definition}

\subsection{多層ネットワーク}

この層を複数重ねることで，より複雑な関数を表現できる．

\begin{definition}{多層ネットワーク}{mlp}
層の数 $L\ge 1$ および各層の次元 $n_0,n_1,\ldots,n_L\in\mathbb{N}$ を与える．
$L$ 個の層 $\ell_{W_l}\colon\mathbb{R}^{n_{l-1}}\to\mathbb{R}^{n_l}$（$l=1,\ldots,L$）
に対し，多層ネットワークを層の合成写像
\[
f = \ell_{W_L}\circ\cdots\circ\ell_{W_1}
\colon\mathbb{R}^{n_0}\to\mathbb{R}^{n_L}
\]
によって定める．すなわち，$h_0=x$ として
\[
h_l = \ell_{W_l}(h_{l-1}) = \sigma(W_l^\top h_{l-1})
\qquad (l=1,\ldots,L)
\]
と逐次的に計算し，$f(x)=h_L$ を出力とする．
\end{definition}

\begin{example}{$L=2$（ReLU + シグモイド）}{two_layer}
$n_0=3,\, n_1=4,\, n_2=1$ とする．

入力 $x\in\mathbb{R}^3$ に対し
\[
\left\{

h_1 & \ell_{W_1}(x) = \mathrm{ReLU}(W_1^\top x) \in \mathbb{R}^4, \\

\end{array}
\right.
\]
となり，$f(x)=h_2$ を出力とする．
ここで $W_1\in\mathbb{R}^{3\times 4}$，
$W_2\in\mathbb{R}^{4\times 1}$ である．
\end{example}

\section{損失関数と勾配降下法}

ニューラルネットワークの学習は，モデルの出力と目標値の乖離を測る損失関数を最小化する問題に帰着される．
本節では損失関数を定義し，これを最小化するための勾配降下法を導入する．
ここではまず，重み行列に制約を課さない場合を扱う．

\subsection{損失関数}

\begin{definition}{コスト関数と損失関数}{cost}
予測値 $\hat{y}\in\mathbb{R}^{p}$ と目標値 $y\in\mathbb{R}^{p}$ の乖離を測る
滑らかな関数 $c\colon\mathbb{R}^{p}\times\mathbb{R}^{p}\to\mathbb{R}$ を
コスト関数と呼ぶ．
訓練データ $\{(x_i,\, y_i)\}_{i=1}^{N}\subset\mathbb{R}^{n}\times\mathbb{R}^{p}$
に対し，損失関数 $\mathcal{L}$ を
\[
\mathcal{L}(W) = \frac{1}{N}\sum_{i=1}^{N} c\bigl(f_W(x_i),\, y_i\bigr)
\]
によって定める．これを最小化するような重み行列 $W$ を求めたい：
$\displaystyle\min_{W}\, \mathcal{L}(W)$．
\end{definition}

\begin{example}{平均二乗誤差}{mse}
$c(\hat{y},\,y)=\|\hat{y}-y\|^2$ とすると，損失関数は
\[
\mathcal{L}(W) = \frac{1}{N}\sum_{i=1}^{N}\bigl\|f_W(x_i)-y_i\bigr\|^2
\]
となる．これは回帰問題における代表的な損失関数である．
\end{example}

\subsection{ユークリッド勾配}

損失関数を最小化するには，$\mathcal{L}$ が最も急速に減少する方向を知る必要がある．

\begin{definition}{ユークリッド勾配}{egrad}
フロベニウス内積 $\langle A,B\rangle=\mathrm{tr}(A^\top B)$ に関して，
\[
d\mathcal{L}(W)[Z] = \langle \nabla \mathcal{L}(W),\, Z \rangle
\qquad (\forall\, Z\in\mathbb{R}^{n\times p})
\]
を満たす行列 $\nabla \mathcal{L}(W)\in\mathbb{R}^{n\times p}$ を
$\mathcal{L}$ の $W$ におけるユークリッド勾配と呼ぶ．
\end{definition}

\subsection{勾配降下法}

制約のない場合，損失関数 $\mathcal{L}$ の最小化にはユークリッド勾配を用いた
勾配降下法が基本的な手法となる．

\begin{definition}{勾配降下法}{gd}
初期値 $W_0\in\mathbb{R}^{n\times p}$ および学習率 $\eta>0$ を与える．
反復
\[
W_{k+1} = W_k - \eta\,\nabla \mathcal{L}(W_k)
\qquad (k=0,1,2,\ldots)
\]
を勾配降下法と呼ぶ．
\end{definition}

しかし，$W$ に制約を課す場合，勾配降下法では更新後の $W$ が制約を満たす保証がない．
この問題は第3節で導入する多様体上の最適化により解決される．

\subsection{誤差逆伝播法}

多層ネットワークでは重み行列が $L$ 個存在し，各 $W_l$ に関する勾配を個別に計算する必要がある．
連鎖律を用いることで，出力層から入力層へ向かって効率的にすべての勾配を計算できる．

多層ネットワーク $f=\ell_{W_L}\circ\cdots\circ\ell_{W_1}$ において，
各重み行列 $W_l$ に関する偏微分 $\partial\mathcal{L}/\partial W_l$ を
連鎖律により計算する方法を述べる．

$a_l=W_l^\top h_{l-1}$ とおくと $h_l=\sigma(a_l)$ であるから，
合成関数の微分の連鎖が次のように整理される．

\begin{proposition}{誤差逆伝播法}{backprop}
訓練データ $(x,y)$ に対する損失を $\mathcal{L}=c(h_L,\,y)$ とする（$c$ はコスト関数）．
$\delta_l = \dfrac{\partial\mathcal{L}}{\partial a_l}\in\mathbb{R}^{n_l}$ とおくと，
連鎖律から
\[
\delta_L = \sigma'(a_L)\odot\frac{\partial\mathcal{L}}{\partial h_L},
\qquad
\delta_l = \sigma'(a_l)\odot(W_{l+1}\,\delta_{l+1})
\quad (l=L{-}1,\ldots,1)
\]
が成り立つ．ここで $\odot$ は要素ごとの積である．
さらに，$a_l=W_l^\top h_{l-1}$ の微分から
\[
\frac{\partial\mathcal{L}}{\partial W_l} = h_{l-1}\,\delta_l^\top
\in\mathbb{R}^{n_{l-1}\times n_l}
\qquad (l=1,\ldots,L)
\]
を得る．
\end{proposition}

$l=L$ から $l=1$ へ逆向きに $\delta_l$ を計算することで，
全ての $\partial\mathcal{L}/\partial W_l$ が得られる．
計算量は順伝播と同程度であり，層数 $L$ に対して線形である．

\section{多様体上の最適化}

\subsection{制約付き最適化と多様体}

最適化変数に制約を課す場合を考える．
$\mathbb{R}^d$ の部分集合 $\mathcal{M}$ 上で目的関数 $f$ を最小化する問題
\[
\min_{x \in \mathcal{M}}\, f(x)
\]
を制約付き最適化と呼ぶ．
制約が滑らかな等式 $F(x) = 0$（$F\colon \mathbb{R}^d \to \mathbb{R}^m$）で定められ，
$F$ の微分 $dF(x)$ が制約面上の各点で全射であるとき，
$\mathcal{M} = F^{-1}(0)$ は $\mathbb{R}^d$ の滑らかな部分多様体をなす．
例えば，単位球面 $S^{n-1} = \{x \in \mathbb{R}^n \mid \|x\|^2 = 1\}$ は
$F(x) = \|x\|^2 - 1$ の零集合として定まる多様体である．

多様体上の最適化では，ユークリッド空間の勾配降下法に対応する概念を
多様体上に拡張する必要がある．
以下では，そのために必要な道具立て——接空間，リーマン勾配，レトラクション——を順に導入する．

\subsection{接空間}

多様体 $\mathcal{M}$ 上の点 $x$ において，$\mathcal{M}$ に沿って動ける方向の集合を接空間と呼ぶ．

\begin{definition}{接空間}{tangent_general}
多様体 $\mathcal{M}$ 上の点 $x$ における接空間 $T_x\mathcal{M}$ を，
$\mathcal{M}$ 上の滑らかな曲線 $x(t)$（$x(0)=x$）に対する
速度ベクトル $\dot{x}(0)$ の全体として定義する．
\end{definition}

滑らかな写像 $F\colon\mathbb{R}^d\to\mathbb{R}^m$ に対して
$\mathcal{M}=\{x\in\mathbb{R}^d\mid F(x)=0\}$ と表される場合，
$dF(x)$ が全射ならば $T_x\mathcal{M} = \ker dF(x)$ が成り立つ．
直感的には，接空間は $x$ における多様体の「接平面」に相当する．

\subsection{リーマン勾配}

ユークリッド勾配 $\nabla f(x)$ は $\mathbb{R}^d$ 全体における最急降下方向であるが，
多様体上で動ける方向は接空間 $T_x\mathcal{M}$ に限られる．
ユークリッド勾配には接空間の法線方向の成分が含まれており，
この成分は多様体上での $f$ の変化に寄与しない．
リーマン勾配はユークリッド勾配から法線成分を除いたものであり，
多様体上での正しい最急降下方向を与える．

\begin{definition}{リーマン勾配}{rgrad}
多様体 $\mathcal{M}$ 上の点 $x$ において，
\[
df(x)[z] = \langle \mathrm{grad}\, f(x),\, z \rangle
\qquad (\forall\, z \in T_x\mathcal{M})
\]
を満たす $\mathrm{grad}\, f(x) \in T_x\mathcal{M}$ を
$f$ の $x$ におけるリーマン勾配と呼ぶ．
\end{definition}

リーマン勾配は，ユークリッド勾配の接空間 $T_x\mathcal{M}$ への直交射影として計算できる．
具体的な射影の形は多様体ごとに異なり，接空間の構造に依存する．

\subsection{レトラクション}

リーマン勾配の方向に $x$ を更新しても，$x - \eta\,\mathrm{grad}\, f(x)$ は
一般に $\mathcal{M}$ 上に留まらない．
そこで，接空間上の更新を多様体上に引き戻す写像が必要となる．

\begin{definition}{レトラクション}{retraction}
多様体 $\mathcal{M}$ 上の点 $x$ において，
滑らかな写像 $R_x\colon T_x\mathcal{M}\to\mathcal{M}$ が
以下の 2 条件を満たすとき，$R$ をレトラクションと呼ぶ：
\[
R_x(0_x) = x, \qquad
\frac{d}{dt}R_x(tZ)\bigg|_{t=0} = Z
\quad (\forall\, Z\in T_x\mathcal{M}).
\]
\end{definition}

条件 1 はゼロ方向への更新が現在の点を保つことを，
条件 2 は微小な更新が接ベクトルの方向に沿うことを要請する．

\subsection{リーマン勾配降下法}

以上の道具立てを組み合わせると，多様体上の勾配降下法が次のように定式化される．
初期値 $x_0 \in \mathcal{M}$ および学習率 $\eta > 0$ に対し，反復
\[
x_{k+1} = R_{x_k}\bigl(-\eta\,\mathrm{grad}\, f(x_k)\bigr)
\qquad (k = 0, 1, 2, \ldots)
\]
をリーマン勾配降下法と呼ぶ．
各反復ではリーマン勾配で降下方向を定め，レトラクションにより多様体上の点に戻す．
具体的なアルゴリズムの構成には，対象とする多様体ごとに
接空間・リーマン勾配・レトラクションを特定する必要がある．
次節では Stiefel 多様体を例にとり，これらを具体的に計算する．

%%  第4節：Stiefel 多様体への適用

\section{Stiefel 多様体による制約}

第3節の一般的枠組みを，ニューラルネットワークの重み行列の直交性制約に適用する．
本節では Stiefel 多様体上で接空間・リーマン勾配・レトラクションを具体的に計算し，
アルゴリズムを構成する．

\subsection{Stiefel 多様体}

重み行列 $W$ に直交性制約 $W^\top W=I_p$ を課すことを考える．
この制約は $W$ の特異値がすべて $1$ であることと同値であり，
$f_W$ による変換のスケールが一定に保たれる．

\begin{definition}{Stiefel 多様体}{stiefel}
$p\le n$ を仮定する．
Stiefel 多様体 $\mathrm{St}(p,n)$ を
\[
\mathrm{St}(p,n)
=
\{\, W\in\mathbb{R}^{n\times p}\mid W^\top W = I_p \,\}
\]
によって定義する．
\end{definition}

この制約のもとで，本稿では次の最適化問題を考える．
\[
\min_{W\in\mathrm{St}(p,n)}\, \mathcal{L}(W).
\]

以降，第4.5節で多層ネットワークに拡張するまでは，
簡単のため単一の重み行列 $W\in\mathbb{R}^{n\times p}$ に対する最適化を扱う．

\subsection{接空間}

第3節で定義した接空間を Stiefel 多様体上で特徴づける．

\begin{proposition}{}{tangent}
$W\in\mathrm{St}(p,n)$ における接空間は
\[
T_W\mathrm{St}(p,n)
=
\{\, Z\in\mathbb{R}^{n\times p}
\mid
W^\top Z + Z^\top W = 0
\,\}
\]
で与えられる．
\end{proposition}

\begin{proof}
$\mathcal{T}=\{\, Z\in\mathbb{R}^{n\times p}\mid W^\top Z + Z^\top W = 0\,\}$ とおく．

($T_W\mathrm{St}(p,n)\subset\mathcal{T}$)\\
$\mathrm{St}(p,n)$ 上の滑らかな曲線 $W(t)$ が $W(0)=W$ を満たすとする．
制約 $W(t)^\top W(t)=I_p$ の両辺を $t=0$ で微分すると
$\dot W(0)^\top W + W^\top\dot W(0)=0$
を得る．すなわち $\dot W(0)\in\mathcal{T}$ である．

($\mathcal{T}\subset T_W\mathrm{St}(p,n)$)\\
制約写像 $F\colon\mathbb{R}^{n\times p}\to\mathrm{Sym}(p)=\{\, S\in\mathbb{R}^{p\times p}\mid S=S^\top \,\}$ を
$F(W)=W^\top W - I_p$ で定める．
$F$ の微分は
\[
dF(W)[Z]=W^\top Z + Z^\top W
\]
で与えられる．任意の $S\in\mathrm{Sym}(p)$ に対し $Z=\tfrac{1}{2}WS$ とおくと
\[
dF(W)\bigl[\tfrac{1}{2}WS\bigr]
= \tfrac{1}{2}(W^\top W S + S W^\top W)
= S
\]
となるから，$dF(W)$ は全射である．
したがって $\mathrm{St}(p,n)=F^{-1}(0)$ は正則値の逆像であり，
$T_W\mathrm{St}(p,n)=\ker dF(W)=\mathcal{T}$ が成り立つ．
\end{proof}

\subsection{リーマン勾配}

第3節で定義したリーマン勾配を Stiefel 多様体上で具体的に計算する．

\begin{remark}{直交分解}{orthogonal_decomp}
内積空間において，ノルム $1$ の元 $u$ が与えられたとき，
任意の元 $v$ は法線成分 $\langle u,v\rangle\, u$ と
接線成分 $v - \langle u,v\rangle\, u$ に直交分解できる．
この分解はベクトルに限らず，
フロベニウス内積を備えた行列空間など一般の内積空間で成り立つ．
\end{remark}

\begin{proposition}{リーマン勾配の計算}{rgrad_formula}
リーマン勾配は，ユークリッド勾配 $G=\nabla \mathcal{L}(W)$ を用いて
\[
\mathrm{grad}\,\mathcal{L}(W)
=
G - W\,\mathrm{sym}(W^\top G)
\]
で与えられる．ここで $\mathrm{sym}(A)=\tfrac{1}{2}(A+A^\top)$ である．
\end{proposition}

\begin{proof}
$G$ を接空間成分 $Z$ と法線成分 $WS$（$S$ は対称行列）に分解する：
\[
G=Z+WS.
\]
$W^\top W=I_p$ より $W^\top G = W^\top Z + S$ である．
接空間条件 $W^\top Z + Z^\top W=0$ から
$W^\top Z$ は歪対称であり，
\[
S
= \mathrm{sym}(W^\top G)
\]
と定まる．
したがって $Z=G-W\,\mathrm{sym}(W^\top G)$ を得る．
\end{proof}

\begin{example}{単位円上のリーマン勾配}{rgrad_circle}
$n=2,\, p=1$ の場合，$\mathrm{St}(1,2)$ は単位円 $S^1$ である．
$\mathcal{L}(w)=(w_1)^2$ とし，
$a=\begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}\in S^1$
における勾配を考える．各成分で偏微分すると，ユークリッド勾配は
\[
\nabla \mathcal{L}(w)
=\begin{pmatrix}\partial \mathcal{L}/\partial w_1\\\partial \mathcal{L}/\partial w_2\end{pmatrix}
=\begin{pmatrix}2w_1\\0\end{pmatrix}
\]
であるから，$w=a$ では
$\nabla \mathcal{L}(a)=(\sqrt{2},\,0)^\top$ となる．
$a$ における接空間は
$T_{a} S^1 = \{z\in\mathbb{R}^2 \mid a^\top z = 0\}$
であり，ユークリッド勾配 $\nabla \mathcal{L}(a)$ は接空間に属さない．
実際，$a^\top \nabla \mathcal{L}(a) = \frac{1}{\sqrt{2}}\cdot\sqrt{2} = 1 \neq 0$ である．

注意~\ref{rem:orthogonal_decomp}の直交分解を
$\nabla \mathcal{L}(a)$ に適用すると，
\[
\underbrace{\begin{pmatrix}\sqrt{2}\\0\end{pmatrix}}_{\nabla \mathcal{L}(a)}
=
\underbrace{\begin{pmatrix}1/\sqrt{2}\\-1/\sqrt{2}\end{pmatrix}}_{\text{接線成分}}
+
\underbrace{\begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}}_{\text{法線成分}}
\]
となる．法線成分は円から離れる方向であり，
円上での $\mathcal{L}$ の変化に寄与しない．
したがって，$-\nabla \mathcal{L}(a)$ の方向に動くと円の外に出てしまい，
円上における最急降下方向とはならない．

一方，リーマン勾配は接線成分のみを取り出したものであり，
\[
\mathrm{grad}\,\mathcal{L}(a)
= \nabla \mathcal{L}(a) - a\,(a^\top \nabla \mathcal{L}(a))
= \begin{pmatrix}\sqrt{2}\\0\end{pmatrix}
  - \begin{pmatrix}1/\sqrt{2}\\1/\sqrt{2}\end{pmatrix}
= \begin{pmatrix}1/\sqrt{2}\\-1/\sqrt{2}\end{pmatrix}
\]
となる．これは命題~\ref{prop:rgrad_formula}の公式
$G - W\,\mathrm{sym}(W^\top G)$ において $p=1$ とした場合に一致する．
$-\mathrm{grad}\,\mathcal{L}(a)$ の方向に進めば
円上に沿って $\mathcal{L}$ が減少する．
\end{example}

\subsection{QR レトラクション}

第3節で定義したレトラクションを Stiefel 多様体上で QR 分解により構成する．

\begin{definition}{QR 分解}{qr}
$\mathrm{rank}(A)=p$ を満たす行列 $A\in\mathbb{R}^{n\times p}$ に対し，
\[
A = QR,
\qquad Q\in\mathrm{St}(p,n),\quad R\in\mathbb{R}^{p\times p}
\]
を満たす直交行列 $Q$ と上三角行列 $R$（対角成分が正）の組が一意に存在する．
これを $A$ の QR 分解と呼び，$Q=\mathrm{qf}(A)$ と書く．
\end{definition}

\begin{example}{}{qr_example}
$n=3,\, p=2$ の場合：
\[
\underbrace{
\begin{pmatrix}
1 & 1 \\
1 & 0 \\
0 & 1
\end{pmatrix}
}_{A}
=
\underbrace{
\begin{pmatrix}
\frac{1}{\sqrt{2}} & \frac{1}{\sqrt{6}} \\[4pt]
\frac{1}{\sqrt{2}} & -\frac{1}{\sqrt{6}} \\[4pt]
0 & \frac{2}{\sqrt{6}}
\end{pmatrix}
}_{Q=\mathrm{qf}(A)}
\underbrace{
\begin{pmatrix}
\sqrt{2} & \frac{1}{\sqrt{2}} \\[4pt]
0 & \frac{3}{\sqrt{6}}
\end{pmatrix}
}_{R}.
\]
\end{example}

$\mathrm{qf}(A)\in\mathrm{St}(p,n)$ であるから，
更新後の行列に QR 分解を適用することで多様体上に戻すことができる．
すなわち，$W\in\mathrm{St}(p,n)$ および $Z\in T_W\mathrm{St}(p,n)$ に対し，
\[
R_W(Z) = \mathrm{qf}(W+Z)
\]
は定義~\ref{def:retraction}の意味でレトラクションとなる．
これを QR レトラクションと呼ぶ．

\subsection{アルゴリズム}

これまでに導入した Stiefel 多様体上のリーマン勾配と QR レトラクションを組み合わせることで，
最適化問題
\[
\min_{W\in\mathrm{St}(p,n)}\, \mathcal{L}(W)
\]
を解くことができる．
アルゴリズムを Algorithm~\ref{alg:rsgd} にまとめる．

\begin{algorithm}{alg:rsgd}
\textbf{Stiefel 多様体上のリーマン勾配降下法} \\[4pt]
\textbf{Input:} 損失関数 $\mathcal{L}$，学習率 $\eta > 0$，初期値 $W_0 \in \mathrm{St}(p,n)$ \\[4pt]
\textbf{for} $k = 0, 1, 2, \ldots$ \textbf{do}
\begin{enumerate}
  \setlength{\itemsep}{2pt}
  \item $G_k \gets \nabla \mathcal{L}(W_k)$
    \hfill $\triangleright$ ユークリッド勾配（定義~\ref{def:egrad}）
  \item $\xi_k \gets G_k - W_k\,\mathrm{sym}(W_k^\top G_k)$
    \hfill $\triangleright$ リーマン勾配（命題~\ref{prop:rgrad_formula}）
  \item $W_{k+1} \gets \mathrm{qf}(W_k - \eta\,\xi_k)$
    \hfill $\triangleright$ QR レトラクション（定義~\ref{def:qr}）
\end{enumerate}
\textbf{end for}
\end{algorithm}

各ステップで QR 分解により $W_{k+1}\in\mathrm{St}(p,n)$ が保証されるため，
すべての反復を通じて直交性制約 $W_k^\top W_k = I_p$ が保たれる．

多層ネットワーク $f=\ell_{W_L}\circ\cdots\circ\ell_{W_1}$ において
各層の重み行列に Stiefel 制約 $W_l\in\mathrm{St}(n_l,n_{l-1})$ を課す場合を考える．
$W_l\in\mathbb{R}^{n_{l-1}\times n_l}$ に対し
$W_l^\top W_l = I_{n_l}$ が成り立つためには $n_{l-1}\ge n_l$ が必要であるから，
各層の次元は $n_0\ge n_1\ge\cdots\ge n_L$ を満たすとする．

誤差逆伝播（命題~\ref{prop:backprop}）で得られるユークリッド勾配
$G_l=\partial\mathcal{L}/\partial W_l$ に対し，
リーマン勾配と QR レトラクションを各層で独立に適用する．
手順を Algorithm~\ref{alg:rsgd_ml} にまとめる．

\begin{algorithm}{alg:rsgd_ml}
\textbf{多層ネットワークのリーマン勾配降下法} \\[4pt]
\textbf{Input:} コスト関数 $c$，学習率 $\eta > 0$，
初期値 $W_l^{(0)} \in \mathrm{St}(n_l,n_{l-1})$（$l=1,\ldots,L$） \\[4pt]
\textbf{for} $k = 0, 1, 2, \ldots$ \textbf{do}
\begin{enumerate}
  \setlength{\itemsep}{2pt}
  \item 順伝播：$h_0=x$，\;$h_l = \sigma(W_l^{(k)\top} h_{l-1})$
    \hfill（$l=1,\ldots,L$）
  \item 誤差逆伝播：$G_l = \partial\mathcal{L}/\partial W_l$
    を命題~\ref{prop:backprop}により計算
    \hfill（$l=L,\ldots,1$）
  \item リーマン勾配：$\xi_l = G_l - W_l^{(k)}\,\mathrm{sym}(W_l^{(k)\top} G_l)$
    \hfill（$l=1,\ldots,L$）
  \item 更新：$W_l^{(k+1)} = \mathrm{qf}(W_l^{(k)} - \eta\,\xi_l)$
    \hfill（$l=1,\ldots,L$）
\end{enumerate}
\textbf{end for}
\end{algorithm}

以下に $L=2$ の具体例を示す．

\begin{example}{$L=2$ の学習手順}{full_example}
$n_0=4,\, n_1=3,\, n_2=1$，
$\sigma_1=\mathrm{ReLU}$，$\sigma_2=\mathrm{sigmoid}$，
$c(\hat{y},y)=(\hat{y}-y)^2$ とする．
$W_1\in\mathrm{St}(3,4)$，$W_2\in\mathrm{St}(1,3)$ であり，
訓練データ $(x,y)\in\mathbb{R}^4\times\mathbb{R}$ に対する
1 回の更新は以下の通りである．

\begin{center}
\begin{tikzpicture}[
  neuron/.style={circle, draw, thick, minimum size=14pt, inner sep=0pt},
  hidden/.style={neuron},
  output/.style={neuron, fill=orange!15},
]
  \def\ls{2.5}

  % Input layer (4 nodes)
  \foreach \i in {1,...,4}
    \node[neuron] (i\i) at (0, {2.5-\i}) {};

  % Hidden layer (3 nodes)
  \foreach \i in {1,2,3}
    \node[hidden] (h\i) at (\ls, {2-\i}) {};

  % Output layer (1 node)
  \node[output] (o1) at (2*\ls, 0) {};

  % Connections: input -> hidden
  \foreach \i in {1,...,4}
    \foreach \j in {1,2,3}
      \draw[-, gray!70] (i\i) -- (h\j);

  % Connections: hidden -> output
  \foreach \i in {1,2,3}
    \draw[-, gray!70] (h\i) -- (o1);

  % Layer labels
  \node[font=\small] at (0, -2.0) {$h_0 = x \in\mathbb{R}^4$};
  \node[font=\small] at (\ls, -1.5) {$h_1 \in\mathbb{R}^3$};
  \node[font=\small] at (2*\ls, -0.8) {$h_2 \in\mathbb{R}$};

  % Weight labels
  \node[font=\small] at (0.5*\ls, 2.0) {$W_1$};
  \node[font=\small] at (1.5*\ls, 1.5) {$W_2$};

  % Activation labels
  \node[font=\scriptsize, blue!60!black] at (\ls, 2.5) {ReLU};
  \node[font=\scriptsize, orange!60!black] at (2*\ls, 0.8) {sigmoid};

\end{tikzpicture}
\end{center}

\textbf{1. 順伝播：}
\[
h_1 = \mathrm{ReLU}(W_1^\top x)\in\mathbb{R}^3,\qquad
h_2 = \mathrm{sigmoid}(W_2^\top h_1)\in\mathbb{R}.
\]

\textbf{2. 損失の計算：}
$\mathcal{L} = (h_2-y)^2$．

\textbf{3. 誤差逆伝播：}
$a_l = W_l^\top h_{l-1}$ とおくと，
\begin{align*}
\delta_2 &= \mathrm{sigmoid}'(a_2)\cdot 2(h_2-y) \\
         &= h_2(1-h_2)\cdot 2(h_2-y)\in\mathbb{R}, \\[6pt]
\delta_1 &= \mathrm{ReLU}'(a_1)\odot(W_2\,\delta_2) \\
         &= \mathbf{1}_{a_1>0}\odot(W_2\,\delta_2)\in\mathbb{R}^3.
\end{align*}
ここで $\mathbf{1}_{a_1>0}$ は $a_1$ の各成分が正なら $1$，
それ以外なら $0$ をとるベクトルである
（例~\ref{ex:activation_examples}）．
ユークリッド勾配は
$G_1 = x\,\delta_1^\top\in\mathbb{R}^{4\times 3}$，\;
$G_2 = h_1\,\delta_2^\top\in\mathbb{R}^{3\times 1}$ である．

\textbf{4. リーマン勾配：}
\[
\xi_l = G_l - W_l\,\mathrm{sym}(W_l^\top G_l)
\qquad (l=1,2).
\]

\textbf{5. 更新：}
\[
W_l \gets \mathrm{qf}(W_l - \eta\,\xi_l)\in\mathrm{St}
\qquad (l=1,2).
\]
\end{example}

以上により，多層ネットワークの各層の重み行列に直交性制約を課しながら
学習を行うアルゴリズムが構成された．
各更新ステップで QR レトラクションを適用することで，
すべての反復を通じて制約 $W_l^\top W_l = I_{n_l}$ が保たれる．
