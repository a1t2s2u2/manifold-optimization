\section{深層学習における最適化問題}

本節では，以降の議論に必要な基本的枠組みを定める．

\begin{definition}[線形変換モデル]
$n,p\in\mathbb{N}$ とする．
重み行列 $W\in\mathbb{R}^{n\times p}$ および
入力 $x\in\mathbb{R}^{n}$ に対し，
写像 $f_W\colon\mathbb{R}^{n}\to\mathbb{R}^{p}$ を
\[
f_W(x)=W^\top x
\]
によって定める．
\end{definition}

滑らかな関数 $L\colon\mathbb{R}^{n\times p}\to\mathbb{R}$ を損失関数と呼ぶ．これを最小化するような重み行列 $W$ を求めたい．
\[
\min_{W\in\mathbb{R}^{n\times p}}\, L(W)
\]

深層学習では，一般的に勾配降下法を用いてこの最適化問題を解く．

\begin{definition}[ユークリッド勾配]
フロベニウス内積 $\langle A,B\rangle=\mathrm{tr}(A^\top B)$ に関して，
\[
dL(W)[Z] = \langle \nabla L(W),\, Z \rangle
\qquad (\forall\, Z\in\mathbb{R}^{n\times p})
\]
を満たす行列 $\nabla L(W)\in\mathbb{R}^{n\times p}$ を
$L$ の $W$ におけるユークリッド勾配と呼ぶ．
\end{definition}

\begin{definition}[勾配降下法]
初期値 $W_0\in\mathbb{R}^{n\times p}$ および学習率 $\eta>0$ を与える．
反復
\[
W_{k+1} = W_k - \eta\,\nabla L(W_k)
\qquad (k=0,1,2,\ldots)
\]
を勾配降下法と呼ぶ．
\end{definition}

\section{Stiefel 多様体上の最適化}

本節では，重み行列 $W$ に直交性制約を課した最適化問題を考える．
$W$ の列ベクトルが正規直交系をなすことを要求することで，
特徴量の冗長性を排除し，学習の安定性を高めることができる．

\begin{definition}[Stiefel 多様体]
$p\le n$ を満たす自然数 $n,p$ に対し，
Stiefel 多様体 $\mathrm{St}(p,n)$ を
\[
\mathrm{St}(p,n)
=
\{\, W\in\mathbb{R}^{n\times p}\mid W^\top W = I_p \,\}
\]
によって定義する．
\end{definition}

本稿で扱う最適化問題は次で与えられる．

\begin{definition}[制約付き最適化問題]
\[
\min_{W\in\mathrm{St}(p,n)}\, L(W).
\]
\end{definition}

\begin{remark}
$\mathrm{St}(p,n)$ は $\mathbb{R}^{n\times p}$ の閉部分多様体であり，
通常の勾配降下法による更新 $W-\eta\,\nabla L(W)$ は
一般に $\mathrm{St}(p,n)$ 上に留まらない．
したがって，多様体の幾何構造を考慮した最適化手法が必要となる．
\end{remark}

\subsection{接空間}

\begin{proposition}
$W\in\mathrm{St}(p,n)$ における接空間は
\[
T_W\mathrm{St}(p,n)
=
\{\, Z\in\mathbb{R}^{n\times p}
\mid
W^\top Z + Z^\top W = 0
\,\}
\]
で与えられる．
\end{proposition}

\begin{proof}
$\mathrm{St}(p,n)$ 上の滑らかな曲線 $W(t)$ に対し，
制約 $W(t)^\top W(t)=I_p$ を $t$ で微分すると
$\dot W^\top W + W^\top\dot W=0$
を得る．$Z=\dot W$ とおけば結論が従う．
\end{proof}

\subsection{リーマン勾配}

\begin{proposition}\label{prop:rgrad}
ユークリッド勾配 $G=\nabla L(W)$ の
$T_W\mathrm{St}(p,n)$ への直交射影は
\[
\mathrm{grad}\,L(W)
=
G - W\,\mathrm{sym}(W^\top G)
\]
で与えられる．ここで $\mathrm{sym}(A)=\tfrac{1}{2}(A+A^\top)$ である．
これを $W$ におけるリーマン勾配と呼ぶ．
\end{proposition}

\begin{proof}
$G$ を接空間成分 $Z$ と法線成分 $WS$（$S$ は対称行列）に分解する：
\[
G=Z+WS.
\]
$W^\top W=I_p$ より $W^\top G = W^\top Z + S$ である．
接空間条件 $W^\top Z + Z^\top W=0$ から
$W^\top Z$ は歪対称であり，
\[
S
= \mathrm{sym}(W^\top G)
\]
と定まる．
したがって $Z=G-W\,\mathrm{sym}(W^\top G)$ を得る．
\end{proof}

\subsection{レトラクション}

\begin{definition}[QR レトラクション]
$W\in\mathrm{St}(p,n)$ および $Z\in T_W\mathrm{St}(p,n)$ に対し，
レトラクション $R_W\colon T_W\mathrm{St}(p,n)\to\mathrm{St}(p,n)$ を
\[
R_W(Z) = \mathrm{qf}(W+Z)
\]
によって定める．
ここで $\mathrm{qf}$ は QR 分解における直交因子を表す．
\end{definition}

\subsection{アルゴリズム}

以上を組み合わせることで，
Stiefel 多様体上の勾配降下法が次のように得られる．

学習率 $\eta>0$ とし，初期値 $W_0\in\mathrm{St}(p,n)$ を与える．
各反復 $k=0,1,2,\ldots$ において：
\[
\begin{aligned}
G_k &= \nabla L(W_k),\\
\xi_k &= G_k - W_k\,\mathrm{sym}(W_k^\top G_k),\\
W_{k+1} &= \mathrm{qf}(W_k - \eta\,\xi_k).
\end{aligned}
\]
